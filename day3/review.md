  ```js
    // incrementing the counter
    // counter++;
    // counter = counter + 1;

    // iterates - repeats
    while (counter <= 10) {
      console.log(counter);
      counter++;
    }

    for (let counter=0; counter <= 10; counter++) {
      console.log(counter);
    }
  ```
